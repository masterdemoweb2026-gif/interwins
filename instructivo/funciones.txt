get_compatibilidades_accesorio:
  with base as (
    select
      coalesce(
        case when "Nombre del atributo 1"='Compatibilidad' then "Valor(es) del atributo 1" end,
        case when "Nombre del atributo 2"='Compatibilidad' then "Valor(es) del atributo 2" end,
        case when "Nombre del atributo 3"='Compatibilidad' then "Valor(es) del atributo 3" end,
        case when "Nombre del atributo 4"='Compatibilidad' then "Valor(es) del atributo 4" end,
        case when "Nombre del atributo 5"='Compatibilidad' then "Valor(es) del atributo 5" end,
        case when "Nombre del atributo 6"='Compatibilidad' then "Valor(es) del atributo 6" end,
        case when "Nombre del atributo 7"='Compatibilidad' then "Valor(es) del atributo 7" end,
        case when "Nombre del atributo 8"='Compatibilidad' then "Valor(es) del atributo 8" end
      ) as compatibilidad
    from public.inter_products_staging
    where
      coalesce(
        case when "Nombre del atributo 1"='Tipo Producto' then "Valor(es) del atributo 1" end,
        case when "Nombre del atributo 2"='Tipo Producto' then "Valor(es) del atributo 2" end,
        case when "Nombre del atributo 3"='Tipo Producto' then "Valor(es) del atributo 3" end,
        case when "Nombre del atributo 4"='Tipo Producto' then "Valor(es) del atributo 4" end,
        case when "Nombre del atributo 5"='Tipo Producto' then "Valor(es) del atributo 5" end,
        case when "Nombre del atributo 6"='Tipo Producto' then "Valor(es) del atributo 6" end,
        case when "Nombre del atributo 7"='Tipo Producto' then "Valor(es) del atributo 7" end,
        case when "Nombre del atributo 8"='Tipo Producto' then "Valor(es) del atributo 8" end
      ) = 'Accesorios'
      and coalesce(
        case when "Nombre del atributo 1"='Tipo Accesorio' then "Valor(es) del atributo 1" end,
        case when "Nombre del atributo 2"='Tipo Accesorio' then "Valor(es) del atributo 2" end,
        case when "Nombre del atributo 3"='Tipo Accesorio' then "Valor(es) del atributo 3" end,
        case when "Nombre del atributo 4"='Tipo Accesorio' then "Valor(es) del atributo 4" end,
        case when "Nombre del atributo 5"='Tipo Accesorio' then "Valor(es) del atributo 5" end,
        case when "Nombre del atributo 6"='Tipo Accesorio' then "Valor(es) del atributo 6" end,
        case when "Nombre del atributo 7"='Tipo Accesorio' then "Valor(es) del atributo 7" end,
        case when "Nombre del atributo 8"='Tipo Accesorio' then "Valor(es) del atributo 8" end
      ) ilike ('%' || p_tipo_accesorio || '%')
  ),
  split as (
    select trim(x) as compatibilidad
    from base, lateral regexp_split_to_table(base.compatibilidad, ',') as x
  )
  select distinct compatibilidad
  from split
  where compatibilidad is not null and compatibilidad <> ''
  order by compatibilidad;

    select ip.frecuencia
    from public.inter_products ip
    where ip.tipo_producto = p_tipo_producto
      and (p_modalidad is null or ip.modalidad = p_modalidad)
      and (p_portabilidad is null or ip.portabilidad = p_portabilidad)
      and (p_tecnologia is null or ip.tecnologia ilike ('%' || p_tecnologia || '%'))
      and ip.frecuencia is not null
  ),
  split as (
    select trim(x) as frecuencia
    from base,
    lateral regexp_split_to_table(base.frecuencia, ',') as x
  )
  select distinct frecuencia
  from split
  where frecuencia <> ''
  order by frecuencia;

get_modalidades:
  select distinct ip.modalidad
  from public.inter_products ip
  where ip.tipo_producto = p_tipo_producto
    and (p_portabilidad is null or ip.portabilidad = p_portabilidad)
    and (p_tecnologia is null or ip.tecnologia ilike ('%' || p_tecnologia || '%'))
    and ip.modalidad is not null
  order by ip.modalidad;


get_portabilidades:
  select distinct ip.portabilidad
  from public.inter_products ip
  where ip.tipo_producto = p_tipo_producto
    and (p_modalidad is null or ip.modalidad = p_modalidad)
    and (p_tecnologia is null or ip.tecnologia ilike ('%' || p_tecnologia || '%'))
    and ip.portabilidad is not null
  order by ip.portabilidad;


get_productos_filtrados:
select *
from public.inter_products
where tipo_producto = trim(p_tipo_producto)
  and (p_modalidad is null or modalidad = trim(p_modalidad))
  and (p_portabilidad is null or portabilidad = trim(p_portabilidad))
  and (p_frecuencia is null or frecuencia ilike ('%' || trim(p_frecuencia) || '%'))
  and (p_tecnologia is null or tecnologia ilike ('%' || trim(p_tecnologia) || '%'))
  and (p_canales is null or canales = trim(p_canales))
  and (p_consumo is null or consumo ilike ('%' || trim(p_consumo) || '%'));

get_tecnologias:
  with raw as (
    select ip.tecnologia
    from public.inter_products ip
    where ip.tipo_producto = p_tipo_producto
      and ip.tecnologia is not null
  ),
  split as (
    select trim(x) as tecnologia
    from raw,
    lateral regexp_split_to_table(raw.tecnologia, ',') as x
  )
  select distinct tecnologia
  from split
  where tecnologia <> ''
  order by tecnologia;

get_tipos_accesorio:
  with base as (
    select
      coalesce(
        case when "Nombre del atributo 1"='Tipo Accesorio' then "Valor(es) del atributo 1" end,
        case when "Nombre del atributo 2"='Tipo Accesorio' then "Valor(es) del atributo 2" end,
        case when "Nombre del atributo 3"='Tipo Accesorio' then "Valor(es) del atributo 3" end,
        case when "Nombre del atributo 4"='Tipo Accesorio' then "Valor(es) del atributo 4" end,
        case when "Nombre del atributo 5"='Tipo Accesorio' then "Valor(es) del atributo 5" end,
        case when "Nombre del atributo 6"='Tipo Accesorio' then "Valor(es) del atributo 6" end,
        case when "Nombre del atributo 7"='Tipo Accesorio' then "Valor(es) del atributo 7" end,
        case when "Nombre del atributo 8"='Tipo Accesorio' then "Valor(es) del atributo 8" end
      ) as tipo_accesorio
    from public.inter_products_staging
    where
      coalesce(
        case when "Nombre del atributo 1"='Tipo Producto' then "Valor(es) del atributo 1" end,
        case when "Nombre del atributo 2"='Tipo Producto' then "Valor(es) del atributo 2" end,
        case when "Nombre del atributo 3"='Tipo Producto' then "Valor(es) del atributo 3" end,
        case when "Nombre del atributo 4"='Tipo Producto' then "Valor(es) del atributo 4" end,
        case when "Nombre del atributo 5"='Tipo Producto' then "Valor(es) del atributo 5" end,
        case when "Nombre del atributo 6"='Tipo Producto' then "Valor(es) del atributo 6" end,
        case when "Nombre del atributo 7"='Tipo Producto' then "Valor(es) del atributo 7" end,
        case when "Nombre del atributo 8"='Tipo Producto' then "Valor(es) del atributo 8" end
      ) = 'Accesorios'
  ),
  split as (
    select trim(x) as tipo_accesorio
    from base, lateral regexp_split_to_table(base.tipo_accesorio, ',') as x
  )
  select distinct tipo_accesorio
  from split
  where tipo_accesorio is not null and tipo_accesorio <> ''
  order by tipo_accesorio;
